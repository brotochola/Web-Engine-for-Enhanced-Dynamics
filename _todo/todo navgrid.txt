NavGrid:

/**
 * NavGrid
 * -------
 * Representa la grilla de navegación lógica de una escena.
 * 
 * - Inmutable durante el frame
 * - Versionada (cualquier cambio invalida caches)
 * - Usada desde logic workers
 * - Internamente cachea flowfields y paths (A*)
 * - Nunca bloquea ni aloca en hot paths
 */
class NavGrid {

  // =========================================================
  // 🔹 Consultas HOT (por frame, O(1))
  // =========================================================

  /**
   * Devuelve el vector de movimiento sugerido desde (cx, cy)
   * hacia (tx, ty) usando flowfield pathfinding.
   * 
   * - Si el flowfield existe: samplea y devuelve el vector
   * - Si no existe o aún no está listo: devuelve (0,0)
   * - Nunca bloquea
   * - Si es necesario, agenda el cálculo en background
   */
  static requestVector(
    cx: number, cy: number,
    tx: number, ty: number,
    outVec2: Vec2
  ): void


  /**
   * Devuelve la posición absoluta de la siguiente celda del path A*
   * desde (fromX, fromY) hacia (toX, toY).
   * 
   * - Si el path está disponible: outPos = centro de la siguiente celda
   * - Si el path no está listo: outPos = posición actual (fromX, fromY)
   * - Nunca bloquea
   * - Agenda el cálculo si es necesario
   */
  static getNextAStarPosition(
    fromX: number, fromY: number,
    toX: number, toY: number,
    outPos: Vec2
  ): void


  // =========================================================
  // 🔹 Consultas COLD (planificación, debug, IA puntual)
  // =========================================================

  /**
   * Devuelve el path completo A* entre dos posiciones.
   * 
   * - Si el path está cacheado: llena outPath
   * - Si no: outPath queda vacío y se calcula en background
   * - El resultado es determinista mientras la NavGrid no cambie
   */
  static getPathAStar(
    fromX: number, fromY: number,
    toX: number, toY: number,
    outPath: Vec2[]
  ): void


  // =========================================================
  // 🔹 Consultas de grilla / utilidades
  // =========================================================

  /**
   * Devuelve el ID de la celda correspondiente a una posición del mundo.
   */
  static getCellAt(x: number, y: number): number


  /**
   * Indica si una celda es caminable.
   */
  static isCellWalkable(cellId: number): boolean


  /**
   * Indica si una posición del mundo es caminable.
   */
  static isPositionWalkable(x: number, y: number): boolean


  /**
   * Devuelve el centro del mundo correspondiente a una celda.
   */
  static getCellCenter(cellId: number, outPos: Vec2): void


  // =========================================================
  // 🔹 Mutación / rebuild (NO hot path)
  // =========================================================

  /**
   * Reconstruye completamente la NavGrid a partir de entidades
   * estáticas u otra geometría lógica.
   * 
   * - Invalida todos los flowfields y paths cacheados
   * - Incrementa la versión interna
   * - Debe llamarse fuera del hot path
   */
  static rebuild(options: {
    staticEntities?: Entity[]
    agentRadius?: number
    // flags, layers, costos, etc.
  }): void


  /**
   * Invalida manualmente todos los caches de navegación.
   * Útil para cambios pequeños o forzados.
   */
  static invalidate(): void
}

Flowfield → devuelve vectores (masas)

A* → devuelve posiciones (pathfollowing)

Nada bloquea

Nada expone IDs

Nada obliga a saber cómo funciona por dentro
